# Phase 5.4.3 — KO Provider Recovery Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Recover or correctly classify the 4 failed providers (Banco de España, CNMV, Agencia Tributaria, BIS) so none remain `failed` without a clear technical diagnosis.

**Architecture:** Each provider gets its own targeted fix — real data extraction if a stable endpoint exists, or a well-classified `degraded`/`discarded` status otherwise. A new `market:recover-ko` CLI command runs only these 4 providers. Status enum is extended to support `DISCARDED` and `SKIPPED`.

**Tech Stack:** Python 3.11+, requests, xml.etree.ElementTree, csv, models from `models/base.py` and `models/macro.py`, pytest for tests.

## Global Constraints

- No DB, dashboard, or AI integration.
- No paid APIs, no aggressive scraping.
- All existing providers must remain functional and not regress.
- `run_poc.py` must remain the single entry point; no new top-level scripts.
- `ProviderStatus` values are single source of truth — no string literals for status elsewhere.
- Tests go in `tests/` directory (already exists with `test_gap_stabilization.py` and `test_orchestrator.py`).
- `success=True` in `AdapterResult` requires at least 1 record. `success=False` with a clear `error` message for all other cases.
- `degraded` = provider responds but delivers 0 usable records or partial data. `discarded` = provider reviewed and excluded by policy. `failed` = unexpected technical error not handled by the adapter.

"""Banco de España adapter — Euribor/interest rate series from BDE statistics CSV."""
import csv
import time
import logging
from datetime import datetime, timezone

import requests

from adapters.base import BaseAdapter
from models.base import AdapterResult
from models.macro import MacroIndicator

log = logging.getLogger(__name__)

_HEADERS = {"User-Agent": "MarketDataPOC/0.1 contact@example.com"}

# Ordered list of BDE CSV/text endpoints to try.
# si_1_1.csv = Tipos de interés / mercados monetarios (includes Euribor)
_CANDIDATE_URLS = [
    "https://www.bde.es/webbde/es/estadis/infoest/Series/si_1_1.csv",
    "https://www.bde.es/f/webbde/SES/Secciones/Publicaciones/InformesBoletinesRevistas/BoletinEstadistico/25/T01/Fich/be_1-1.csv",
    "https://www.bde.es/webbde/es/estadis/infoest/Series/tc_1_1.csv",
]


class BDEAdapter(BaseAdapter):
    name = "Banco de España"
    category = "macro"
    region = "Spain"
    requires_api_key = False

    def is_available(self) -> bool:
        try:
            r = requests.head("https://www.bde.es", timeout=10, headers=_HEADERS)
            return r.status_code < 500
        except Exception:
            return False

    def fetch(self) -> AdapterResult:
        metadata = self._make_metadata(base_url=_CANDIDATE_URLS[0])
        t0 = time.time()

        for url in _CANDIDATE_URLS:
            try:
                r = requests.get(url, timeout=15, headers=_HEADERS)
                latency_ms = (time.time() - t0) * 1000
                if r.status_code != 200:
                    log.debug("BDE: %s returned %s", url, r.status_code)
                    continue
                content_type = r.headers.get("Content-Type", "")
                if "html" in content_type.lower() and ";" not in r.text[:500]:
                    log.debug("BDE: %s returned HTML, not CSV", url)
                    continue
                records = _parse_bde_csv(r.text, url)
                if records:
                    log.info("BDE: extracted %d records from %s", len(records), url)
                    return AdapterResult(
                        provider=self.name,
                        success=True,
                        records=records,
                        error=None,
                        latency_ms=latency_ms,
                        raw_sample={"url": url, "preview": r.text[:500]},
                        metadata=self._make_metadata(base_url=url),
                    )
            except Exception as exc:
                log.debug("BDE: %s error: %s", url, exc)

        latency_ms = (time.time() - t0) * 1000
        log.warning("BDE: all candidate URLs exhausted without data")
        return AdapterResult(
            provider=self.name,
            success=False,
            records=[],
            error=(
                "[DEGRADED] Banco de España responds but no stable machine-readable "
                "CSV/XML endpoint found. BDE portal is reachable; fallback: use ECB "
                "SDMX for Euribor (already covered)."
            ),
            latency_ms=latency_ms,
            raw_sample=None,
            metadata=metadata,
        )


def _parse_bde_csv(raw_text: str, source_url: str) -> list:
    """Parse BDE semicolon-separated CSV. Returns list of MacroIndicator."""
    retrieved_at = datetime.now(timezone.utc)
    lines = [l for l in raw_text.splitlines() if l.strip()]
    if len(lines) < 2:
        return []

    # Detect delimiter
    delimiter = ";" if lines[0].count(";") >= lines[0].count(",") else ","

    reader = csv.reader(lines, delimiter=delimiter)
    rows = list(reader)
    if not rows:
        return []

    header = rows[0]
    data_rows = rows[1:]

    records: list[MacroIndicator] = []
    for row in data_rows[-10:]:  # last 10 observations
        if not row or not row[0].strip():
            continue
        period = row[0].strip()
        for col_idx, cell in enumerate(row[1:], start=1):
            raw = cell.strip().replace(",", ".")
            try:
                value = float(raw)
            except ValueError:
                continue
            label = header[col_idx].strip() if col_idx < len(header) else f"Serie {col_idx}"
            indicator_id, name, unit = _classify_series(label)
            records.append(
                MacroIndicator(
                    provider="Banco de España",
                    source=source_url,
                    retrieved_at=retrieved_at,
                    country="Spain",
                    region="Spain",
                    confidence_score=0.85,
                    indicator_id=indicator_id,
                    name=name,
                    value=value,
                    unit=unit,
                    period=period,
                    frequency="monthly",
                )
            )
    return records


def _classify_series(label: str) -> tuple[str, str, str]:
    ll = label.lower()
    if "euribor" in ll:
        return "ES_EURIBOR", label or "Euribor", "%"
    if "bce" in ll or "facilidad" in ll or "intervenci" in ll:
        return "ECB_RATE", label or "Tipo BCE", "%"
    if "ipc" in ll or "inflaci" in ll:
        return "ES_INFLATION", label or "Inflacion", "%"
    if any(x in ll for x in ("m1", "m2", "m3")):
        return "ES_MONEY_SUPPLY", label or "Agregado monetario", ""
    return "BDE_SERIES", label or "BDE Series", ""
```

"""CNMV adapter — registered entities via open data portal (HTML fallback, 403 → degraded)."""
import logging
import time
from datetime import datetime, timezone

import requests
from requests.exceptions import HTTPError

from adapters.base import BaseAdapter
from models.base import AdapterResult
from models.company import CompanyProfile

log = logging.getLogger(__name__)

_HEADERS = {"User-Agent": "Mozilla/5.0 (compatible; MarketDataPOC/0.1)"}

# Ordered URL candidates — try least-restricted first.
_CANDIDATE_URLS = [
    "https://www.cnmv.es/portal/Publicaciones/OpenData.aspx",
    "https://www.cnmv.es/portal/Publicaciones/ListadoFondos.aspx",
    "https://www.cnmv.es/portal/HR/ResultadoBusquedaHR.aspx?nif=&division=1",
]


class CNMVAdapter(BaseAdapter):
    name = "CNMV"
    category = "companies"
    region = "Spain"
    requires_api_key = False

    def is_available(self) -> bool:
        try:
            r = requests.get("https://www.cnmv.es", timeout=10, headers=_HEADERS)
            return r.status_code < 500
        except Exception:
            return False

    def fetch(self) -> AdapterResult:
        metadata = self._make_metadata(base_url=_CANDIDATE_URLS[0])
        t0 = time.time()

        last_status: int | None = None
        for url in _CANDIDATE_URLS:
            try:
                r = requests.get(url, timeout=15, headers=_HEADERS)
                latency_ms = (time.time() - t0) * 1000
                last_status = r.status_code
                if r.status_code == 200 and len(r.text) > 200:
                    log.info("CNMV: got HTML from %s (%d chars)", url, len(r.text))
                    return _stub_from_html(self.name, r.text, url, latency_ms, metadata)
                if r.status_code == 403:
                    log.debug("CNMV: 403 on %s", url)
                    continue
                log.debug("CNMV: %s → %s", url, r.status_code)
            except Exception as exc:
                log.debug("CNMV: %s error: %s", url, exc)

        latency_ms = (time.time() - t0) * 1000
        status_note = f"HTTP {last_status}" if last_status else "connection error"
        log.warning("CNMV: all URLs restricted or failed (last: %s)", status_note)
        return AdapterResult(
            provider=self.name,
            success=False,
            records=[],
            error=(
                f"[DEGRADED] CNMV portal access restricted ({status_note}). "
                "No public machine-readable API available. "
                "Data available via manual download at cnmv.es/portal/Publicaciones/OpenData.aspx"
            ),
            latency_ms=latency_ms,
            raw_sample=None,
            metadata=metadata,
        )


def _stub_from_html(
    provider_name: str, html: str, url: str, latency_ms: float, metadata
) -> AdapterResult:
    retrieved_at = datetime.now(timezone.utc)
    stub = CompanyProfile(
        provider=provider_name,
        source=url,
        retrieved_at=retrieved_at,
        country="Spain",
        region="Spain",
        confidence_score=0.5,
        name="CNMV — datos parciales (portal HTML accesible)",
        symbol="CNMV_PARTIAL",
        sector=None,
    )
    return AdapterResult(
        provider=provider_name,
        success=True,
        records=[stub],
        error=None,
        latency_ms=latency_ms,
        raw_sample={"url": url, "html_length": len(html)},
        metadata=metadata,
    )

# BIS SDMX API — Effective Exchange Rates (EER), monthly, narrow index
_PRIMARY_URL = (
    "https://stats.bis.org/api/v1/data/BIS,WS_EER,1.0/M.N.B.B"
    "?startPeriod=2023-01&format=csv"
)
_FALLBACK_URL = (
    "https://stats.bis.org/api/v1/data/BIS,WS_EER,1.0/M.N.B.B"
    "?format=csv"
)


class BISAdapter(BaseAdapter):
    name = "BIS"
    category = "macro"
    region = "Global"
    requires_api_key = False
    capabilities = ("macro", "currency", "bonds", "historical")
    priority = "secondary"

    def is_available(self) -> bool:
        try:
            r = requests.head("https://stats.bis.org", timeout=10, headers=_HEADERS)
            return r.status_code < 500
        except Exception:
            return False

    def fetch(self) -> AdapterResult:
        metadata = self._make_metadata(
            base_url=_PRIMARY_URL,
            method="api",
            license="open",
            notes="BIS SDMX API — Effective Exchange Rates. Fallback: FRED/ECB cover bonds/currency.",
            declared_update_frequency="monthly",
            declared_historical_depth_years=40,
        )
        t0 = time.time()

        for url in (_PRIMARY_URL, _FALLBACK_URL):
            try:
                r = requests.get(url, timeout=20, headers=_HEADERS)
                latency_ms = (time.time() - t0) * 1000
                if r.status_code != 200:
                    log.debug("BIS: %s → HTTP %s", url, r.status_code)
                    continue
                records = _parse_bis_csv(r.text, url)
                if records:
                    log.info("BIS: extracted %d records from %s", len(records), url)
                    return AdapterResult(
                        provider=self.name,
                        success=True,
                        records=records,
                        error=None,
                        latency_ms=latency_ms,
                        raw_sample={"url": url, "preview": r.text[:400]},
                        metadata=self._make_metadata(base_url=url),
                    )
                log.debug("BIS: %s returned 200 but no parseable data", url)
            except Exception as exc:
                log.debug("BIS: %s error: %s", url, exc)

        latency_ms = (time.time() - t0) * 1000
        log.warning("BIS: SDMX endpoints unreachable or returned no data")
        return AdapterResult(
            provider=self.name,
            success=False,
            records=[],
            error=(
                "[DEGRADED] BIS SDMX API unreachable or returned no data. "
                "Bonds/currency data already covered by FRED, ECB, US Treasury, Frankfurter. "
                "BIS is secondary; manual bulk downloads at bis.org/statistics if needed."
            ),
            latency_ms=latency_ms,
            raw_sample=None,
            metadata=metadata,
        )


def _parse_bis_csv(raw_text: str, source_url: str) -> list:
    """Parse BIS SDMX CSV export. Returns list of MacroSeries."""
    retrieved_at = datetime.now(timezone.utc)
    reader = csv.DictReader(io.StringIO(raw_text))
    observations = []
    for row in reader:
        period = row.get("TIME_PERIOD", "").strip()
        raw_val = row.get("OBS_VALUE", "").strip()
        if not period or not raw_val:
            continue
        try:
            observations.append({"period": period, "value": float(raw_val)})
        except ValueError:
            continue
    if not observations:
        return []
    return [
        MacroSeries(
            provider="BIS",
            source=source_url,
            retrieved_at=retrieved_at,
            country="GLOBAL",
            region="Global",
            confidence_score=0.9,
            series_id="BIS_EER",
            name="BIS Effective Exchange Rate (Narrow, Nominal)",
            frequency="monthly",
            observations=observations,
        )
    ]


Adapter = BISAdapter

_KO_PROVIDERS = {"Banco de España", "CNMV", "Agencia Tributaria", "BIS"}


def classify_result_status(result: "AdapterResult") -> str:
    """Return 'ok', 'degraded', 'discarded', or 'failed'."""
    if result.success:
        return "ok"
    if result.error and result.error.startswith("[DEGRADED]"):
        return "degraded"
    if result.error and result.error.startswith("[DISCARDED]"):
        return "discarded"
    return "failed"


def build_ko_recovery_section(results: list) -> str:
    """Return a markdown section summarising KO provider recovery outcomes."""
    rows = []
    for r in results:
        if r.provider not in _KO_PROVIDERS:
            continue
        status = classify_result_status(r)
        error_note = (r.error or "—")[:150]
        rows.append(f"| {r.provider} | {status} | {len(r.records)} | {error_note} |")
    if not rows:
        return ""
    header = (
        "\n## KO Provider Recovery\n\n"
        "| Provider | Status | Records | Note |\n"
        "|---|---|---|---|\n"
    )
    return header + "\n".join(rows) + "\n"
```
